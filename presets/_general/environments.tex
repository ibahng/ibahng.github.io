
% tcolorbox package
\usepackage{tcolorbox}
\tcbuselibrary{breakable}

\definecolor{Yellow}{HTML}{D9A300} % Optimized Gold/Ochre

\definecolor{Green}{HTML}{008000}  % Office Green

  colframe = black,
  colback  = black!1,
  coltitle = black,
  colbacktitle = black!15,
  breakable, 
  arc=0mm,
  boxrule=0.7pt,
  toptitle=3pt,
  bottomtitle=3pt,
  left=8pt,
  right=8pt,
  top=6pt,
  bottom=6pt,
  before skip=12pt,
  after skip=12pt,
  bottomrule at break=-1pt,
  toprule at break=-1pt,
  fonttitle=\bfseries,
  fontupper=\setlength{\parskip}{0.6em}
}

% Custom Environments
\newtcolorbox[auto counter, number within=section]{definition}[2][]
{
  title = \textbf{Definition \thetcbcounter: ~#2},
  #1
}

\newtcolorbox[auto counter, number within=section]{core}[2][]
{
  title = \textbf{Core Concept \thetcbcounter: ~#2},
  #1,
}
